–ù–∞ –≤—Ç–æ—Ä–æ–º –¥–Ω–µ –Ω–∞—à–µ–≥–æ –∫—É—Ä—Å–∞ –º—ã –ø—Ä–æ–¥–æ–ª–∂–∏–º –∏–∑—É—á–µ–Ω–∏–µ JavaScript, —Å–æ—Å—Ä–µ–¥–æ—Ç–æ—á–∏–≤—à–∏—Å—å –Ω–∞ —Ü–∏–∫–ª–∞—Ö, –±—É–ª–µ–≤–æ–π –∞–ª–≥–µ–±—Ä–µ –∏ —Å—Å—ã–ª–æ—á–Ω—ã—Ö —Ç–∏–ø–∞—Ö –¥–∞–Ω–Ω—ã—Ö. –≠—Ç–∏ —Ç–µ–º—ã —è–≤–ª—è—é—Ç—Å—è –≤–∞–∂–Ω—ã–º–∏ –¥–ª—è –ø–æ–Ω–∏–º–∞–Ω–∏—è –±–æ–ª–µ–µ —Å–ª–æ–∂–Ω—ã—Ö –∫–æ–Ω—Ü–µ–ø—Ü–∏–π –ø—Ä–æ–≥—Ä–∞–º–º–∏—Ä–æ–≤–∞–Ω–∏—è –∏ —ç—Ñ—Ñ–µ–∫—Ç–∏–≤–Ω–æ–≥–æ –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏—è —è–∑—ã–∫–∞.

# –¶–∏–∫–ª—ã
–¶–∏–∫–ª—ã –ø–æ–∑–≤–æ–ª—è—é—Ç –ø–æ–≤—Ç–æ—Ä—è—Ç—å –≤—ã–ø–æ–ª–Ω–µ–Ω–∏–µ –±–ª–æ–∫–∞ –∫–æ–¥–∞ –¥–æ —Ç–µ—Ö –ø–æ—Ä, –ø–æ–∫–∞ –≤—ã–ø–æ–ª–Ω—è–µ—Ç—Å—è –æ–ø—Ä–µ–¥–µ–ª–µ–Ω–Ω–æ–µ —É—Å–ª–æ–≤–∏–µ. –í JavaScript —Å—É—â–µ—Å—Ç–≤—É–µ—Ç –Ω–µ—Å–∫–æ–ª—å–∫–æ —Ç–∏–ø–æ–≤ —Ü–∏–∫–ª–æ–≤: `for`, `while` –∏ `do...while` (*–ø–æ—Å–ª–µ–¥–Ω–∏–π –º—ã –≤ —Ü–µ–ª–æ–º –∑–∞—Ç—Ä–∞–≥–∏–≤–∞—Ç—å –Ω–µ –±—É–¥–µ–º*).

## –¶–∏–∫–ª —Å –∏–Ω–∏—Ü–∏–∞–ª–∏–∑–∞—Ü–∏–µ–π —Å—á–µ—Ç—á–∏–∫–∞ (for)
**–¶–∏–∫–ª `for`** –∏—Å–ø–æ–ª—å–∑—É–µ—Ç—Å—è, –∫–æ–≥–¥–∞ –∏–∑–≤–µ—Å—Ç–Ω–æ –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ –∏—Ç–µ—Ä–∞—Ü–∏–π, –∞ —Ç–∞–∫–∂–µ –≤ —Å–ª—É—á–∞—è—Ö, –∫–æ–≥–¥–∞ –Ω–∞–º –Ω—É–∂–Ω–∞ –∫–æ–º–ø–∞–∫—Ç–Ω–∞—è –∑–∞–ø–∏—Å—å
–¥–ª—è –ø–µ—Ä–µ–º–µ–Ω–Ω–æ–π, –∫–æ—Ç–æ—Ä–∞—è –±—É–¥–µ—Ç –∏–∑–º–µ–Ω—è—Ç—å—Å—è –ø–æ—Å–ª–µ –∫–∞–∂–¥–æ–π –∏—Ç–µ—Ä–∞—Ü–∏–∏ (–ø—Ä–æ—Ö–æ–∂–¥–µ–Ω–∏—è —Ü–∏–∫–ª–∞).

```javascript
for (let i = 0; i < 5; i++) {
    console.log(i); // –í—ã–≤–æ–¥: 0 1 2 3 4
}
```

–°–∞–º —Ü–∏–∫–ª –º–æ–∂–Ω–æ —Ä–∞–∑–¥–µ–ª–∏—Ç—å –Ω–∞ —Ç—Ä–∏ —á–∞—Å—Ç–∏:
1. –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∞—Ü–∏—è —Å—á–µ—Ç—á–∏–∫–∞: `let i = 0;`
2. –£—Å–ª–æ–≤–∏–µ: `i < 5;`
3. –û–±–Ω–æ–≤–ª–µ–Ω–∏–µ —Å—á–µ—Ç—á–∏–∫–∞: `i++`

> –î–≤–∞ –ø–ª—é—Å–∞, –∫–æ—Ç–æ—Ä—ã–µ –∏–¥—É—Ç –≤ –æ–±–Ω–æ–≤–ª–µ–Ω–∏–∏ —Å—á–µ—Ç—á–∏–∫–∞ –Ω–∞–∑—ã–≤–∞—é—Ç—Å—è **–∏–Ω–∫—Ä–µ–º–µ–Ω—Ç–æ–º** (`i++`), –≤—Å–µ —á—Ç–æ –¥–µ–ª–∞–µ—Ç –¥–∞–Ω–Ω—ã–π –æ–ø–µ—Ä–∞—Ç–æ—Ä ‚Äî –ø—Ä–æ—Å—Ç–æ –¥–æ–±–∞–≤–ª—è–µ—Ç –µ–¥–∏–Ω–∏—Ü—É –∫ –ø–µ—Ä–µ–º–µ–Ω–Ω–æ–π —É–∫–∞–∑–∞–Ω–æ–π –¥–æ –∏–Ω–∫—Ä–µ–º–µ–Ω—Ç–∞.
>
> –°—Ç–æ–∏—Ç —É–ø–æ–º—è–Ω—É—Ç—å, —á—Ç–æ —Ç–∞–∫–∂–µ —Å—É—â–µ—Å—Ç–≤—É–µ—Ç –æ–±—Ä–∞—Ç–Ω—ã–π –æ–ø–µ—Ä–∞—Ç–æ—Ä ‚Äî **–¥–µ–∫—Ä–µ–º–µ–Ω—Ç**, –æ–Ω —É–º–µ–Ω—å—à–∞–µ—Ç –ø–µ—Ä–µ–º–µ–Ω–Ω—É—é –Ω–∞ –µ–¥–∏–Ω–∏—Ü—É (`i--`).

## –¶–∏–∫–ª while
–¶–∏–∫–ª while –≤—ã–ø–æ–ª–Ω—è–µ—Ç—Å—è –¥–æ —Ç–µ—Ö –ø–æ—Ä, –ø–æ–∫–∞ —É—Å–ª–æ–≤–∏–µ –∏—Å—Ç–∏–Ω–Ω–æ. –°—Ç–æ–∏—Ç –∑–∞–º–µ—Ç–∏—Ç—å, —á—Ç–æ –ø–µ—Ä–µ–º–µ–Ω–Ω–∞—è —É —ç—Ç–æ–≥–æ —Ü–∏–∫–ª–∞
–æ–±—ä—è–≤–ª—è–µ—Ç—Å—è –µ—â–µ –¥–æ –Ω–∞—á–∞–ª–∞ —Ü–∏–∫–ª–∞ (—Å–º–æ—Ç—Ä–∏—Ç–µ –≥–¥–µ —Ä–∞—Å–ø–æ–ª–æ–∂–µ–Ω–æ `let i` –≤ –ø—Ä–∏–º–µ—Ä–µ –Ω–∏–∂–µ).

–¢–∞–∫–∂–µ, –≤–∞–∂–Ω–æ —É–ø–æ–º—è–Ω—É—Ç—å —á—Ç–æ –≤ —Å–ª—É—á–∞–µ —Å `while` –º—ã —Å–∞–º–∏ –¥–æ–ª–∂–Ω—ã —É–≤–µ–ª–∏—á–∏—Ç—å –∑–Ω–∞—á–µ–Ω–∏–µ —Å—á–µ—Ç—á–∏–∫–∞ –≤–Ω—É—Ç—Ä–∏ —Ü–∏–∫–ª–∞.


```javascript
let i = 0;
while (i < 5) {
    console.log(i); // –í—ã–≤–æ–¥: 0 1 2 3 4
    i++;
}
```

## –¶–∏–∫–ª do...while
–¶–∏–∫–ª do...while –≥–∞—Ä–∞–Ω—Ç–∏—Ä—É–µ—Ç, —á—Ç–æ –±–ª–æ–∫ –∫–æ–¥–∞ –±—É–¥–µ—Ç –≤—ã–ø–æ–ª–Ω–µ–Ω —Ö–æ—Ç—è –±—ã –æ–¥–∏–Ω —Ä–∞–∑, –ø—Ä–µ–∂–¥–µ —á–µ–º –ø—Ä–æ–≤–µ—Ä–∏—Ç—å —É—Å–ª–æ–≤–∏–µ.

```javascript
let i = 0;
do {
    console.log(i); // –í—ã–≤–æ–¥: 0 1 2 3 4
    i++;
} while (i < 5);
```


# –ë—É–ª–µ–≤–∞ –∞–ª–≥–µ–±—Ä–∞
–ë—É–ª–µ–≤–∞ –∞–ª–≥–µ–±—Ä–∞ –≤–∫–ª—é—á–∞–µ—Ç –≤ —Å–µ–±—è –æ–ø–µ—Ä–∞—Ü–∏–∏ AND (&&), OR (||) –∏ NOT (!), –∫–æ—Ç–æ—Ä—ã–µ –∏—Å–ø–æ–ª—å–∑—É—é—Ç—Å—è –¥–ª—è —Å–æ–∑–¥–∞–Ω–∏—è —Å–ª–æ–∂–Ω—ã—Ö —É—Å–ª–æ–≤–∏–π –∏ —É–ø—Ä–∞–≤–ª–µ–Ω–∏—è –ø–æ—Ç–æ–∫–æ–º –≤—ã–ø–æ–ª–Ω–µ–Ω–∏—è –ø—Ä–æ–≥—Ä–∞–º–º—ã.


```javascript
let a = true;
let b = false;

console.log(a && b); // –í—ã–≤–æ–¥: false (AND)
console.log(a || b); // –í—ã–≤–æ–¥: true (OR)
console.log(!a); // –í—ã–≤–æ–¥: false (NOT)
```

# –°—Å—ã–ª–æ—á–Ω—ã–µ —Ç–∏–ø—ã –¥–∞–Ω–Ω—ã—Ö
**–°—Å—ã–ª–æ—á–Ω—ã–µ —Ç–∏–ø—ã –¥–∞–Ω–Ω—ã—Ö**, —Ç–∞–∫–∏–µ –∫–∞–∫ –æ–±—ä–µ–∫—Ç—ã –∏ –º–∞—Å—Å–∏–≤—ã, —Ö—Ä–∞–Ω—è—Ç —Å—Å—ã–ª–∫—É –Ω–∞ –¥–∞–Ω–Ω—ã–µ, –∞ –Ω–µ —Å–∞–º–∏ –¥–∞–Ω–Ω—ã–µ. –≠—Ç–æ –æ–∑–Ω–∞—á–∞–µ—Ç, —á—Ç–æ –ø—Ä–∏ –∫–æ–ø–∏—Ä–æ–≤–∞–Ω–∏–∏ —Å—Å—ã–ª–æ—á–Ω–æ–≥–æ —Ç–∏–ø–∞ –¥–∞–Ω–Ω—ã—Ö, –∏–∑–º–µ–Ω–µ–Ω–∏—è –≤ –æ–¥–Ω–æ–π –ø–µ—Ä–µ–º–µ–Ω–Ω–æ–π –±—É–¥—É—Ç –æ—Ç—Ä–∞–∂–∞—Ç—å—Å—è –∏ –≤ –¥—Ä—É–≥–æ–π.

**–ú–∞—Å—Å–∏–≤—ã** ‚Äî —ç—Ç–æ —É–ø–æ—Ä—è–¥–æ—á–µ–Ω–Ω—ã–µ –∫–æ–ª–ª–µ–∫—Ü–∏–∏ –∑–Ω–∞—á–µ–Ω–∏–π, –∫–æ—Ç–æ—Ä—ã–µ –º–æ–≥—É—Ç –±—ã—Ç—å —Ä–∞–∑–Ω—ã—Ö —Ç–∏–ø–æ–≤.

